73static void MedirDistancia(void *pvParameter)

151static void PrenderLEDS(void *pvParameter)
152{
153 while (true)
154 {
155 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
156 if (ON)
157 {
158 if (distancia < 10)
159 {
160 LedsOffAll();
161 }
162 else if (distancia < 20 && distancia >= 10)
163 {
164 LedOn(LED_1);
165 LedOff(LED_2);
166 LedOff(LED_3);
167 }
168 else if (distancia < 30 && distancia >= 20)
169 {
170 LedOn(LED_1);
171 LedOn(LED_2);
172 LedOff(LED_3);
173 }
174 else if (distancia >= 30)
175 {
176 LedOn(LED_1);

235 xTaskCreate(&PrenderLEDS, "PRENDER", 512, NULL, 5, &PrenderLEDS_task_handle);
236
237 /* Inicialización del conteo de timers */
238 TimerStart(Func_timerB.timer);
