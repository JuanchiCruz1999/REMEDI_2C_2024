112void medir_nivel_agua_task(void *pvParameter){
113 while(1){
114
115 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
116
117 float altura_agua;
118 altura_agua=30-HcSr04ReadDistanceInCentimeters();//el sensor está en la parte de arriba del tanque de 30 cm de altura

182void EncenderYDetenerPorTecla()
183{
184 while(1)
185 {
186 teclas = SwitchesRead();
187
188 if(teclas == SWITCH_1 && ENCENDIDO){
189 ENCENDIDO=!ENCENDIDO;//lo pone en FALSE
190 LedOff(LED_1);
191 GPIOOff(GPIO_ELECTROVALVULA);
192 GPIOOff(GPIO_COMPUERTA_ALIMENTO);
193
194 }
195
196 else if(teclas == SWITCH_1 && !ENCENDIDO){
197 ENCENDIDO=ENCENDIDO;//lo pone en TRUE
198 LedOn(LED_1);
199 GPIOOn(GPIO_ELECTROVALVULA);
200 GPIOOn(GPIO_COMPUERTA_ALIMENTO);
201 }
202 vTaskDelay(CONFIG_BLINK_PERIOD_TECLAS / portTICK_PERIOD_MS);
203 }
204}

211void app_main(void){
212
213 //Inicializo GPIOs
214 GPIOInit(GPIO_ELECTROVALVULA, GPIO_OUTPUT);
215 GPIOInit(GPIO_COMPUERTA_ALIMENTO, GPIO_OUTPUT);
216
217 //Inicializo Sensor de ultrasonido
218 HcSr04Init(GPIO_20, GPIO_21);
219
220 // configuracion para entrada analogica
221 analog_input_config_t analogInput1 = {
222 .input = CH1, // Se configura para leer del canal 1 del conversor analógico-digital (ADC)
223 .mode = ADC_SINGLE, // Se configura para realizar una única lectura analógica
224 };
225 AnalogInputInit(&analogInput1); // Inicializa
226
227 // Para el puerto serie
228 serial_config_t my_uart = {
229 .port = UART_PC,
230 .baud_rate = 115200,
231 .func_p = NULL,
232 .param_p = NULL
233 };
234 UartInit(&my_uart);
235
236 //Para la tecla
237 SwitchesInit();
238 LedsInit();
239 SwitchActivInt(SWITCH_1, &EncenderYDetenerPorTecla, NULL);
240
241 // configuro timer1 para:
242 // 1) control de volumen de agua en los tanques
243 // 2) medir peso de alimento con la balanza
244 // 3) informar por UART las medidas tomadas en 1) y 2)
245 // Las tres tareas se hacen cada 5 segundos
246 timer_config_t timer_1 = {
247 .timer = TIMER_A,
248 .period = PERIODO_TIEMPO_1,
249 .func_p = Timer_1_Func,
250 .param_p = NULL
251 };
252
253 TimerInit(&timer_1); // inicializo timer 1
254 TimerStart(timer_1.timer); // para que comience el timer 1
255
256 //creo tareas
257 xTaskCreate(&medir_nivel_agua_task, "medir volumen de agua", 512, NULL, 5, &task_handle_agua);
258 xTaskCreate(&medir_y_convertir_a_gramos_Task, "pesar alimento", 512, NULL, 5, &task_handle_alimento);
259 xTaskCreate(&informar_por_UART_Task, "mandar mensaje por UART", 512, NULL, 5, &task_handle_UART);
260}
